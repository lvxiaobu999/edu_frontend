# 项目目录规划

src/
├── api/ # 🌐 统一 API 请求层
│ ├── request.ts # Axios 实例封装（含双 Token 无感刷新拦截器）
│ ├── auth.ts # 登录、注册、注销接口
│ ├── users.ts # 用户列表、审核接口
│ ├── dicts.ts # 字典拉取（班级/科目/教研组/学期）
│ └── ... # 其他业务接口
├── assets/ # 🎨 静态资源（图片、全局 CSS/SCSS）
├── components/ # 🧩 全局公共组件
│ ├── DataTable/ # 统一封装的分页数据表格
│ └── RoleGuard/ # 角色权限包装组件
├── layouts/ # 🪟 页面布局
│ └── MainLayout/ # 核心后台布局（侧边栏菜单 + 顶部导航 + 内容区）
├── pages/ # 📄 核心业务视图（与一级菜单严格对应）
│ ├── Dashboard/ # 仪表盘模块
│ ├── Users/ # 用户管理模块
| | ├── tearchs # 学生模块
| | ├── students # 老师模块
| | ├── admin # 管理员模块
│ ├── Dicts/ # 字典管理模块
| | ├── classes # 班级字典
| | ├── research-group # 教研组字典
| | ├── semester # 学期字典
| | ├── subject # 科目字典
│ ├── Exams/ # 考试模块
│ ├── Scores/ # 成绩模块
│ └── Profile/ # 个人档案模块
├── router/ # 🛣️ 路由配置
│ ├── index.tsx # 核心路由表
│ └── AuthRoute.tsx # 动态路由守卫（根据 Token 和 Role 拦截）
├── store/ # 📦 全局状态管理 (Zustand / Redux)
│ └── authStore.ts # 存储用户信息、Token 及当前 Role
├── types/ # 🏷️ TypeScript 类型定义 (如使用 TS)
│ └── api.d.ts # 统一响应结构类型定义
└── utils/ # 🛠️ 工具函数（日期格式化、权限校验等）
